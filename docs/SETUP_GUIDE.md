### Windows PC

```bash
# Download from https://ollama.ai
# After installing, pull the model:
ollama pull llama3

# Test it works:
ollama run llama3
# Type "hello" — if it responds, you're good. Press CTRL+D to exit.
```


## STEP 2: Firebase Setup

1. Go to https://console.firebase.google.com
2. Create a new project called "JARVIS"
3. Enable **Authentication** → Sign-in providers → Enable **Email/Password** and **Google**
4. Go to Project Settings → Service Accounts → Generate new private key
5. Save the JSON file as `backend/firebase-credentials.json`

## STEP 8: Flutter main.dart

```dart
import 'package:flutter/material.dart';
import 'package:firebase_core/firebase_core.dart';
import 'package:provider/provider.dart';
import 'firebase_options.dart';
import 'services/auth_service.dart';
import 'services/api_service.dart';
import 'services/websocket_service.dart';
import 'screens/login_screen.dart';
import 'screens/home_screen.dart';

void main() async {
  WidgetsFlutterBinding.ensureInitialized();
  await Firebase.initializeApp(options: DefaultFirebaseOptions.currentPlatform);
  runApp(const JarvisApp());
}

class JarvisApp extends StatelessWidget {
  const JarvisApp({super.key});

  @override
  Widget build(BuildContext context) {
    return MultiProvider(
      providers: [
        ChangeNotifierProvider(create: (_) => AuthService()),
        Provider(create: (_) => ApiService()),
        Provider(create: (_) => WebSocketService()),
      ],
      child: MaterialApp(
        title: 'JARVIS',
        debugShowCheckedModeBanner: false,
        theme: ThemeData.dark().copyWith(
          colorScheme: const ColorScheme.dark(
            primary: Color(0xFF00E5FF),
            secondary: Color(0xFF0066FF),
          ),
          scaffoldBackgroundColor: const Color(0xFF010A12),
        ),
        home: const AuthGate(),
      ),
    );
  }
}

class AuthGate extends StatelessWidget {
  const AuthGate({super.key});

  @override
  Widget build(BuildContext context) {
    final auth = context.watch<AuthService>();
    return auth.isLoggedIn ? const HomeScreen() : const LoginScreen();
  }
}
```

## STEP 9: API Service (Flutter)

```dart
// services/api_service.dart
import 'package:dio/dio.dart';
import 'package:firebase_auth/firebase_auth.dart';

class ApiService {
  static const baseUrl = 'http://YOUR_PC_IP:8000';  // Change to your PC IP
  late final Dio _dio;

  ApiService() {
    _dio = Dio(BaseOptions(baseUrl: baseUrl));
    _dio.interceptors.add(InterceptorsWrapper(
      onRequest: (options, handler) async {
        final token = await FirebaseAuth.instance.currentUser?.getIdToken();
        if (token != null) {
          options.headers['Authorization'] = 'Bearer $token';
        }
        handler.next(options);
      },
    ));
  }

  Future<List> getReminders() async {
    final res = await _dio.get('/api/reminders');
    return res.data;
  }

  Future<Map> createReminder(String title, DateTime remindAt, {String? desc}) async {
    final res = await _dio.post('/api/reminders', data: {
      'title': title,
      'remind_at': remindAt.toIso8601String(),
      'description': desc ?? '',
    });
    return res.data;
  }

  Future<Map> chat(String message) async {
    final res = await _dio.post('/api/chat', data: {'message': message});
    return res.data;
  }

  Future<List> getNotes() async {
    final res = await _dio.get('/api/notes');
    return res.data;
  }

  Future<Map> createNote(String title, String content) async {
    final res = await _dio.post('/api/notes', data: {'title': title, 'content': content});
    return res.data;
  }

  Future<Map> getDailySummary() async {
    final res = await _dio.get('/api/activity/summary');
    return res.data;
  }

  Future<Map> sendMessage(String platform, String recipient, String message) async {
    final res = await _dio.post('/api/message/send', data: {
      'platform': platform,
      'recipient': recipient,
      'message': message,
    });
    return res.data;
  }

  Future<Map> identifyFace(List<int> imageBytes) async {
    final formData = FormData.fromMap({
      'image': MultipartFile.fromBytes(imageBytes, filename: 'face.jpg'),
    });
    final res = await _dio.post('/api/faces/identify', data: formData);
    return res.data;
  }
}
```

## STEP 10: WebSocket Service (Flutter — real-time)

```dart
// services/websocket_service.dart
import 'dart:convert';
import 'package:web_socket_channel/web_socket_channel.dart';

class WebSocketService {
  static const wsUrl = 'ws://YOUR_PC_IP:8000/ws';
  WebSocketChannel? _channel;

  void connect(int userId) {
    const deviceId = 'android_device_1';
    _channel = WebSocketChannel.connect(
      Uri.parse('$wsUrl/$deviceId/$userId'),
    );
    _channel!.stream.listen(_onMessage, onError: _onError, onDone: _onDone);
    print('WebSocket connected');
  }

  void _onMessage(dynamic raw) {
    final msg = jsonDecode(raw);
    final type = msg['type'];
    final payload = msg['payload'];
    // Handle different message types
    switch (type) {
      case 'chat_response':
        // Notify chat screen
        break;
      case 'face_result':
        // Show face recognition result
        break;
      case 'screen_frame':
        // Update screen share view
        break;
    }
  }

  void send(String type, Map payload) {
    _channel?.sink.add(jsonEncode({'type': type, 'payload': payload}));
  }

  void sendChat(String message) => send('chat', {'text': message});

  void requestScreenShare() => send('screen_request', {'fps': 10});

  void _onError(error) => print('WS Error: $error');
  void _onDone() => print('WS Disconnected');

  void disconnect() => _channel?.sink.close();
}
```

## Troubleshooting

**Ollama not responding:** Make sure `ollama serve` is running in a terminal.

**Vosk model not found:** Check the path in `.env` matches where you extracted it.

**Firebase auth failing:** Make sure the JSON key file path is correct in `.env`.

**Mobile can't reach PC:** Both must be on same WiFi. Disable Windows Firewall temporarily to test, then add a rule for port 8000.

**Face recognition slow:** First run downloads VGG-Face model (~500MB). Subsequent runs use cache.

**WhatsApp QR required:** First run opens Chrome and shows QR. Scan once — the session is saved to `~/.jarvis_browser_profile`.
